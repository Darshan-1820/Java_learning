Classes

User.java 
public class User {
    private String name;            // field
    private String email;

    public User(String name, String email) {  // constructor
        this.name = name;
        this.email = email;
    }

    public String getName() { return name; }           // getter
    public void setName(String name) { this.name = name; } // setter

    @Override
    public String toString() {
        return "User{name='" + name + "', email='" + email + "'}";
    }
}

try-catch Template

Exception Handling 
try {
    // risky code
    Connection conn = DriverManager.getConnection(url, user, pass);
} catch (SQLException e) {
    // handle specific exception
    System.err.println("DB Error: " + e.getMessage());
} catch (Exception e) {
    // catch-all (put specific catches FIRST)
    e.printStackTrace();
} finally {
    // ALWAYS runs (cleanup)
    if (conn != null) conn.close();
}

// try-with-resources (auto-closes)
try (Connection conn = DriverManager.getConnection(url, user, pass);
     PreparedStatement ps = conn.prepareStatement(sql)) {
    // use conn and ps
} // auto-closed here, no finally needed

Collections Cheat Sheet 
// === List ===
List<String> list = new ArrayList<>();
list.add("A");              // append
list.add(0, "B");           // insert at index
list.get(0);                // access by index
list.set(0, "C");           // replace at index
list.remove(0);             // remove by index
list.remove("A");           // remove by value
list.size();                // count
list.contains("A");         // search
list.indexOf("A");          // find position
Collections.sort(list);     // sort

// === Map ===
Map<String, Integer> map = new HashMap<>();
map.put("parcels", 42);    // add/update
map.get("parcels");         // retrieve (null if missing)
map.getOrDefault("x", 0);  // retrieve with fallback
map.containsKey("parcels"); // check key
map.containsValue(42);     // check value
map.remove("parcels");     // delete
map.keySet();               // all keys
map.values();               // all values
map.entrySet();             // all key-value pairs

// === Set ===
Set<String> set = new HashSet<>();
set.add("A");               // add (ignores duplicates)
set.contains("A");          // search
set.remove("A");            // delete

Section 05

SQL Templates

Copy-paste SQL for every common operation.

SQL Quick Reference 
-- CREATE TABLE
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    role VARCHAR(20) DEFAULT 'USER',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE bookings (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    tracking_number VARCHAR(20) UNIQUE NOT NULL,
    status VARCHAR(20) DEFAULT 'PENDING',
    weight DECIMAL(5,2),
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- INSERT
INSERT INTO users (name, email) VALUES ('Darshan', 'darshan@nimbus.com');
INSERT INTO bookings (user_id, tracking_number, weight)
VALUES (1, 'NMB-20260423-001', 2.5);

-- SELECT
SELECT * FROM users;
SELECT name, email FROM users WHERE role = 'ADMIN';
SELECT * FROM bookings WHERE status = 'PENDING' ORDER BY created_at DESC;
SELECT COUNT(*), status FROM bookings GROUP BY status;
SELECT * FROM bookings WHERE weight BETWEEN 1.0 AND 5.0;

-- JOIN
SELECT u.name, b.tracking_number, b.status
FROM users u
INNER JOIN bookings b ON u.id = b.user_id
WHERE b.status = 'DELIVERED';

-- LEFT JOIN (all users, even without bookings)
SELECT u.name, COUNT(b.id) AS total_bookings
FROM users u
LEFT JOIN bookings b ON u.id = b.user_id
GROUP BY u.name;

-- UPDATE
UPDATE bookings SET status = 'SHIPPED' WHERE tracking_number = 'NMB-20260423-001';

-- DELETE
DELETE FROM bookings WHERE status = 'CANCELLED' AND created_at < '2026-01-01';

PreparedStatement — INSERT

Insert with PreparedStatement 
String sql = "INSERT INTO users (name, email, role) VALUES (?, ?, ?)";
try (PreparedStatement ps = conn.prepareStatement(sql)) {
    ps.setString(1, user.getName());
    ps.setString(2, user.getEmail());
    ps.setString(3, user.getRole());
    int rows = ps.executeUpdate();  // returns affected row count
    System.out.println(rows + " row(s) inserted");
}

PreparedStatement — SELECT

Query with ResultSet 
String sql = "SELECT * FROM users WHERE role = ?";
try (PreparedStatement ps = conn.prepareStatement(sql)) {
    ps.setString(1, "ADMIN");
    try (ResultSet rs = ps.executeQuery()) {
        while (rs.next()) {
            int id = rs.getInt("id");
            String name = rs.getString("name");
            String email = rs.getString("email");
            // build User object from these values
        }
    }
}

3incompatible typesWrong type assignment. Cast or fix the type.
