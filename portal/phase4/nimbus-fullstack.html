package com.nimbus.dao;

import com.nimbus.model.User;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.jdbc.core.RowMapper;
import org.springframework.stereotype.Repository;

@Repository
public class UserDAO {

    @Autowired
    private JdbcTemplate jdbcTemplate;

    private final RowMapper<User> userMapper = (rs, rowNum) -> {
        User user = new User();
        user.setUserId(rs.getInt("user_id"));
        user.setName(rs.getString("name"));
        user.setEmail(rs.getString("email"));
        user.setPassword(rs.getString("password"));
        user.setRole(rs.getString("role"));
        return user;
    };

    public User findByEmail(String email) {
        String sql = "SELECT * FROM users WHERE email = ?";
        var results = jdbcTemplate.query(sql, userMapper, email);
        return results.isEmpty() ? null : results.get(0);
    }

    public User findById(int id) {
        String sql = "SELECT * FROM users WHERE user_id = ?";
        var results = jdbcTemplate.query(sql, userMapper, id);
        return results.isEmpty() ? null : results.get(0);
    }

    public int save(User user) {
        String sql = "INSERT INTO users (name, email, password, role) VALUES (?, ?, ?, ?)";
        jdbcTemplate.update(sql, user.getName(), user.getEmail(),
                           user.getPassword(), user.getRole());
        return jdbcTemplate.queryForObject("SELECT LAST_INSERT_ID()", Integer.class);
    }
}
dao/BookingDAO.java 
package com.nimbus.dao;

import com.nimbus.model.Booking;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.jdbc.core.RowMapper;
import org.springframework.stereotype.Repository;
import java.util.List;

@Repository
public class BookingDAO {

    @Autowired
    private JdbcTemplate jdbcTemplate;

    private final RowMapper<Booking> bookingMapper = (rs, rowNum) -> {
        Booking b = new Booking();
        b.setBookingId(rs.getInt("booking_id"));
        b.setUserId(rs.getInt("user_id"));
        b.setSenderName(rs.getString("sender_name"));
        b.setSenderAddress(rs.getString("sender_address"));
        b.setReceiverName(rs.getString("receiver_name"));
        b.setReceiverAddress(rs.getString("receiver_address"));
        b.setWeight(rs.getDouble("weight"));
        b.setStatus(rs.getString("status"));
        b.setPickupDate(rs.getDate("pickup_date").toLocalDate());
        b.setEstimatedDelivery(rs.getDate("estimated_delivery") != null
            ? rs.getDate("estimated_delivery").toLocalDate() : null);
        b.setCreatedAt(rs.getTimestamp("created_at").toLocalDateTime());
        return b;
    };

    public int save(Booking b) {
        String sql = "INSERT INTO bookings (user_id, sender_name, sender_address, "
            + "receiver_name, receiver_address, weight, status, pickup_date, "
            + "estimated_delivery) VALUES (?,?,?,?,?,?,?,?,?)";
        jdbcTemplate.update(sql, b.getUserId(), b.getSenderName(),
            b.getSenderAddress(), b.getReceiverName(), b.getReceiverAddress(),
            b.getWeight(), b.getStatus(), b.getPickupDate(), b.getEstimatedDelivery());
        return jdbcTemplate.queryForObject("SELECT LAST_INSERT_ID()", Integer.class);
    }

    public Booking findById(int id) {
        String sql = "SELECT * FROM bookings WHERE booking_id = ?";
        var results = jdbcTemplate.query(sql, bookingMapper, id);
        return results.isEmpty() ? null : results.get(0);
    }

    public List<Booking> findByUserId(int userId) {
        return jdbcTemplate.query(
            "SELECT * FROM bookings WHERE user_id = ? ORDER BY created_at DESC",
            bookingMapper, userId);
    }

    public List<Booking> findDeliveredByUserId(int userId) {
        return jdbcTemplate.query(
            "SELECT * FROM bookings WHERE user_id = ? AND status = 'DELIVERED' ORDER BY created_at DESC",
            bookingMapper, userId);
    }

    public void updateStatus(int id, String status) {
        jdbcTemplate.update("UPDATE bookings SET status = ? WHERE booking_id = ?", status, id);
    }

    public void updatePickupDate(int id, String pickupDate) {
        jdbcTemplate.update("UPDATE bookings SET pickup_date = ? WHERE booking_id = ?", pickupDate, id);
    }
}

Services

service/UserService.java 
package com.nimbus.service;

import com.nimbus.dao.UserDAO;
import com.nimbus.dto.*;
import com.nimbus.exception.DuplicateEmailException;
import com.nimbus.model.User;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service
public class UserService {

    @Autowired
    private UserDAO userDAO;

    public UserResponse register(RegistrationRequest request) {
        if (userDAO.findByEmail(request.getEmail()) != null) {
            throw new DuplicateEmailException(request.getEmail());
        }
        User user = new User();
        user.setName(request.getName());
        user.setEmail(request.getEmail());
        user.setPassword(request.getPassword());  // Hash in production!
        user.setRole(request.getRole());
        int id = userDAO.save(user);
        user.setUserId(id);
        return new UserResponse(user.getUserId(), user.getName(),
                               user.getEmail(), user.getRole());
    }

    public LoginResponse authenticate(LoginRequest request) {
        User user = userDAO.findByEmail(request.getEmail());
        if (user == null || !user.getPassword().equals(request.getPassword())) {
            throw new RuntimeException("Invalid email or password");
        }
        return new LoginResponse(user.getUserId(), user.getName(),
                                user.getRole(), "Login successful");
    }
}
service/BookingService.java 
package com.nimbus.service;

import com.nimbus.dao.BookingDAO;
import com.nimbus.dto.*;
import com.nimbus.exception.BookingNotFoundException;
import com.nimbus.model.Booking;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.List;
import java.util.stream.Collectors;

@Service
public class BookingService {

    @Autowired
    private BookingDAO bookingDAO;

    public BookingResponse createBooking(BookingRequest request) {
        Booking booking = new Booking();
        booking.setUserId(request.getUserId());
        booking.setSenderName(request.getSenderName());
        booking.setSenderAddress(request.getSenderAddress());
        booking.setReceiverName(request.getReceiverName());
        booking.setReceiverAddress(request.getReceiverAddress());
        booking.setWeight(request.getWeight());
        booking.setStatus("BOOKED");
        booking.setPickupDate(LocalDate.parse(request.getPickupDate()));
        booking.setEstimatedDelivery(LocalDate.parse(request.getPickupDate()).plusDays(3));
        booking.setCreatedAt(LocalDateTime.now());

        int id = bookingDAO.save(booking);
        booking.setBookingId(id);
        return toResponse(booking);
    }

    public BookingResponse findById(int id) {
        Booking booking = bookingDAO.findById(id);
        if (booking == null) throw new BookingNotFoundException(id);
        return toResponse(booking);
    }

    public List<BookingResponse> findByUserId(int userId) {
        return bookingDAO.findByUserId(userId).stream()
            .map(this::toResponse).collect(Collectors.toList());
    }

    public BookingResponse updateStatus(int id, StatusUpdateRequest request) {
        Booking booking = bookingDAO.findById(id);
        if (booking == null) throw new BookingNotFoundException(id);
        bookingDAO.updateStatus(id, request.getStatus());
        booking.setStatus(request.getStatus());
        return toResponse(booking);
    }

    public BookingResponse updatePickup(int id, PickupUpdateRequest request) {
        Booking booking = bookingDAO.findById(id);
        if (booking == null) throw new BookingNotFoundException(id);
        bookingDAO.updatePickupDate(id, request.getPickupDate());
        booking.setPickupDate(LocalDate.parse(request.getPickupDate()));
        return toResponse(booking);
    }

    public List<BookingResponse> getHistory(int userId) {
        return bookingDAO.findDeliveredByUserId(userId).stream()
            .map(this::toResponse).collect(Collectors.toList());
    }

    private BookingResponse toResponse(Booking b) {
        BookingResponse r = new BookingResponse();
        r.setBookingId(b.getBookingId());
        r.setSenderName(b.getSenderName());
        r.setReceiverName(b.getReceiverName());
        r.setWeight(b.getWeight());
        r.setStatus(b.getStatus());
        r.setPickupDate(b.getPickupDate() != null ? b.getPickupDate().toString() : null);
        r.setEstimatedDelivery(b.getEstimatedDelivery() != null ? b.getEstimatedDelivery().toString() : null);
        r.setCreatedAt(b.getCreatedAt());
        return r;
    }
}

package com.nimbus.controller;

import com.nimbus.dto.*;
import com.nimbus.service.UserService;
import jakarta.validation.Valid;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.net.URI;

@RestController
public class UserController {

    @Autowired
    private UserService userService;

    @PostMapping("/api/register")
    public ResponseEntity<UserResponse> register(
            @Valid @RequestBody RegistrationRequest request) {
        UserResponse created = userService.register(request);
        return ResponseEntity.created(
            URI.create("/api/users/" + created.getUserId())
        ).body(created);
    }

    @PostMapping("/api/login")
    public ResponseEntity<LoginResponse> login(
            @Valid @RequestBody LoginRequest request) {
        return ResponseEntity.ok(userService.authenticate(request));
    }
}
controller/BookingController.java 
package com.nimbus.controller;

import com.nimbus.dto.*;
import com.nimbus.service.BookingService;
import jakarta.validation.Valid;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.net.URI;
import java.util.List;

@RestController
@RequestMapping("/api/bookings")
public class BookingController {

    @Autowired
    private BookingService bookingService;

    @PostMapping
    public ResponseEntity<BookingResponse> create(
            @Valid @RequestBody BookingRequest request) {
        BookingResponse created = bookingService.createBooking(request);
        return ResponseEntity.created(
            URI.create("/api/bookings/" + created.getBookingId())
        ).body(created);
    }

    @GetMapping("/{id}")
    public ResponseEntity<BookingResponse> getById(@PathVariable int id) {
        return ResponseEntity.ok(bookingService.findById(id));
    }

    @GetMapping
    public ResponseEntity<List<BookingResponse>> getByUser(@RequestParam int userId) {
        return ResponseEntity.ok(bookingService.findByUserId(userId));
    }

    @PutMapping("/{id}/status")
    public ResponseEntity<BookingResponse> updateStatus(
            @PathVariable int id, @Valid @RequestBody StatusUpdateRequest request) {
        return ResponseEntity.ok(bookingService.updateStatus(id, request));
    }

    @PutMapping("/{id}/pickup")
    public ResponseEntity<BookingResponse> updatePickup(
            @PathVariable int id, @Valid @RequestBody PickupUpdateRequest request) {
        return ResponseEntity.ok(bookingService.updatePickup(id, request));
    }

    @GetMapping("/history")
    public ResponseEntity<List<BookingResponse>> getHistory(@RequestParam int userId) {
        return ResponseEntity.ok(bookingService.getHistory(userId));
    }
}

Services

services/auth.service.ts 
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable, tap } from 'rxjs';
import { environment } from '../../environments/environment';

@Injectable({ providedIn: 'root' })
export class AuthService {
  private apiUrl = environment.apiUrl;

  constructor(private http: HttpClient) {}

  register(data: any): Observable<any> {
    return this.http.post(`${this.apiUrl}/register`, data);
  }

  login(email: string, password: string): Observable<any> {
    return this.http.post(`${this.apiUrl}/login`, { email, password }).pipe(
      tap((response: any) => {
        localStorage.setItem('user', JSON.stringify(response));
      })
    );
  }

  logout(): void {
    localStorage.removeItem('user');
  }

  isLoggedIn(): boolean {
    return localStorage.getItem('user') !== null;
  }

  getUser(): any {
    const user = localStorage.getItem('user');
    return user ? JSON.parse(user) : null;
  }

  getUserRole(): string {
    const user = this.getUser();
    return user ? user.role : '';
  }

  getUserId(): number {
    const user = this.getUser();
    return user ? user.userId : 0;
  }
}

Section 04

How to Run Both

Start the backend on port 8080 and the frontend on port 4200. They talk to each other over HTTP.

Step 1: Start the Backend

Terminal 1 — Spring Boot backend 
cd nimbus-backend
mvn spring-boot:run
# Server starts on http://localhost:8080
# Test: curl http://localhost:8080/api/bookings/1001

Step 2: Start the Frontend

Terminal 2 — Angular frontend 
cd nimbus-frontend
npm install
ng serve
# App opens on http://localhost:4200

Proxy Configuration (Alternative to CORS)

Instead of configuring CORS on the backend, you can proxy Angular's dev server to forward API calls to Spring Boot:

proxy.conf.json — Angular dev proxy 
{
  "/api": {
    "target": "http://localhost:8080",
    "secure": false,
    "changeOrigin": true
  }
}
Start with proxy 
ng serve --proxy-config proxy.conf.json

With the proxy, Angular sends requests to http://localhost:4200/api/... and the dev server forwards them to http://localhost:8080/api/.... No CORS issues because both appear to be on the same origin from the browser's perspective.

In Nimbus

Use both approaches. The proxy for development (no CORS config needed). The WebConfig CORS setup for when Angular is built and deployed separately from the backend (production). In development, update environment.ts to use apiUrl: '/api' (relative, goes through proxy) instead of 'http://localhost:8080/api'.
