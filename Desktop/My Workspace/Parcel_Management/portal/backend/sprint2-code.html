    user_id         VARCHAR(20) PRIMARY KEY,
    first_name      VARCHAR(25),
    last_name       VARCHAR(25),
    full_name       VARCHAR(50),
    email           VARCHAR(100) UNIQUE,
    mobile          VARCHAR(15),
    address         VARCHAR(200),
    password        VARCHAR(30),
    role            VARCHAR(20),
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ===== Booking Table =====
CREATE TABLE IF NOT EXISTS tbl_Booking (
    booking_id          VARCHAR(20) PRIMARY KEY,
    user_id             VARCHAR(20),
    receiver_name       VARCHAR(100),
    receiver_address    VARCHAR(200),
    receiver_mobile     VARCHAR(15),
    weight_gram         DOUBLE,
    contents_description VARCHAR(200),
    delivery_type       VARCHAR(20),
    status              VARCHAR(20) DEFAULT 'Booked',
    service_cost        DOUBLE,
    created_at          TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES tbl_UserProfile(user_id)
);

-- ===== Issue Table =====
CREATE TABLE IF NOT EXISTS tbl_Issue (
    issue_id        VARCHAR(20) PRIMARY KEY,
    user_id         VARCHAR(20),
    booking_id      VARCHAR(20),
    description     VARCHAR(500),
    status          VARCHAR(20) DEFAULT 'Open',
    assigned_to     VARCHAR(20),
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES tbl_UserProfile(user_id)
);

package com.nimbus.util;

import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.concurrent.atomic.AtomicInteger;

public class IdGenerator {

    private static final AtomicInteger userSequence = new AtomicInteger(1);
    private static final AtomicInteger bookingSequence = new AtomicInteger(1);
    private static final AtomicInteger issueSequence = new AtomicInteger(1);

    /**
     * Generate User ID: YYMMDD + initials + sequence
     * Example: 260504DA01
     */
    public static String generateUserId(String firstName, String lastName) {
        String datePart = LocalDate.now().format(DateTimeFormatter.ofPattern("yyMMdd"));
        String initials = ("" + firstName.charAt(0) + lastName.charAt(0)).toUpperCase();
        String seq = String.format("%02d", userSequence.getAndIncrement());
        return datePart + initials + seq;
    }

    /**
     * Generate Booking ID: BK + YYMMDD + sequence
     * Example: BK260504001
     */
    public static String generateBookingId() {
        String datePart = LocalDate.now().format(DateTimeFormatter.ofPattern("yyMMdd"));
        String seq = String.format("%03d", bookingSequence.getAndIncrement());
        return "BK" + datePart + seq;
    }

    /**
     * Generate Issue ID: IS + YYMMDD + sequence
     * Example: IS260504001
     */
    public static String generateIssueId() {
        String datePart = LocalDate.now().format(DateTimeFormatter.ofPattern("yyMMdd"));
        String seq = String.format("%03d", issueSequence.getAndIncrement());
        return "IS" + datePart + seq;
    }
}

package com.nimbus.util;

public class PricingCalculator {

    // Base rates per kg
    private static final double ECONOMY_RATE = 30.0;   // Rs 30 per kg
    private static final double STANDARD_RATE = 50.0;  // Rs 50 per kg
    private static final double EXPRESS_RATE = 80.0;   // Rs 80 per kg

    // Minimum charge
    private static final double MIN_CHARGE = 40.0;

    /**
     * Calculate service cost based on weight (grams) and delivery type.
     * @param weightGram weight in grams
     * @param deliveryType "Economy", "Standard", or "Express"
     * @return calculated cost in rupees
     */
    public static double calculateCost(double weightGram, String deliveryType) {
        double weightKg = weightGram / 1000.0;

        double rate;
        switch (deliveryType.toLowerCase()) {
            case "express":
                rate = EXPRESS_RATE;
                break;
            case "standard":
                rate = STANDARD_RATE;
                break;
            case "economy":
            default:
                rate = ECONOMY_RATE;
                break;
        }

        double cost = weightKg * rate;
        return Math.max(cost, MIN_CHARGE);
    }

    /**
     * Get delivery type options as formatted string for display.
     */
    public static String getDeliveryOptions() {
        return "1. Economy  (Rs 30/kg, 5-7 days)\n" +
               "2. Standard (Rs 50/kg, 3-4 days)\n" +
               "3. Express  (Rs 80/kg, 1-2 days)";
    }
}

package com.nimbus.util;

import java.util.regex.Pattern;

public class InputValidator {

    private static final Pattern EMAIL_PATTERN =
            Pattern.compile("^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$");

    private static final Pattern MOBILE_PATTERN =
            Pattern.compile("^[6-9]\\d{9}$");

    /**
     * Validate email format.
     */
    public static boolean isValidEmail(String email) {
        if (email == null || email.isEmpty()) return false;
        return EMAIL_PATTERN.matcher(email).matches();
    }

    /**
     * Validate Indian mobile number (10 digits, starts with 6-9).
     */
    public static boolean isValidMobile(String mobile) {
        if (mobile == null || mobile.isEmpty()) return false;
        return MOBILE_PATTERN.matcher(mobile).matches();
    }

    /**
     * Validate that name is not empty and has only letters/spaces.
     */
    public static boolean isValidName(String name) {
        if (name == null || name.trim().isEmpty()) return false;
        return name.matches("^[A-Za-z ]+$");
    }

    /**
     * Validate weight is positive and within limit (50kg max).
     */
    public static boolean isValidWeight(double weightGram) {
        return weightGram > 0 && weightGram <= 50000;
    }

    /**
     * Validate password (minimum 6 characters).
     */
    public static boolean isValidPassword(String password) {
        return password != null && password.length() >= 6;
    }

    /**
     * Validate delivery type is one of the allowed values.
     */
    public static boolean isValidDeliveryType(String type) {
        if (type == null) return false;
        String lower = type.toLowerCase();
        return lower.equals("economy") || lower.equals("standard") || lower.equals("express");
    }
}

package com.nimbus.dao;

import com.nimbus.model.User;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.dao.EmptyResultDataAccessException;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.jdbc.core.RowMapper;
import org.springframework.stereotype.Component;

import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.List;

@Component
public class UserDAO {

    @Autowired
    private JdbcTemplate jdbcTemplate;

    // ===== RowMapper: converts a ResultSet row into a User object =====
    private RowMapper<User> userRowMapper = new RowMapper<User>() {
        @Override
        public User mapRow(ResultSet rs, int rowNum) throws SQLException {
            User user = new User();
            user.setUserId(rs.getString("user_id"));
            user.setFirstName(rs.getString("first_name"));
            user.setLastName(rs.getString("last_name"));
            user.setFullName(rs.getString("full_name"));
            user.setEmail(rs.getString("email"));
            user.setMobile(rs.getString("mobile"));
            user.setAddress(rs.getString("address"));
            user.setPassword(rs.getString("password"));
            user.setRole(rs.getString("role"));
            user.setCreatedAt(rs.getTimestamp("created_at"));
            return user;
        }
    };

    /**
     * Find user by ID.
     * @return User or null if not found
     */
    public User findById(String userId) {
        try {
            String sql = "SELECT * FROM tbl_UserProfile WHERE user_id = ?";
            return jdbcTemplate.queryForObject(sql, userRowMapper, userId);
        } catch (EmptyResultDataAccessException e) {
            return null;
        }
    }

    /**
     * Find user by email.
     * @return User or null if not found
     */
    public User findByEmail(String email) {
        try {
            String sql = "SELECT * FROM tbl_UserProfile WHERE email = ?";
            return jdbcTemplate.queryForObject(sql, userRowMapper, email);
        } catch (EmptyResultDataAccessException e) {
            return null;
        }
    }

    /**
     * Get all users.
     */
    public List<User> findAll() {
        String sql = "SELECT * FROM tbl_UserProfile ORDER BY created_at DESC";
        return jdbcTemplate.query(sql, userRowMapper);
    }

    /**
     * Save a new user to the database.
     */
    public void save(User user) {
        String sql = "INSERT INTO tbl_UserProfile (user_id, first_name, last_name, full_name, " +
                     "email, mobile, address, password, role) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)";
        jdbcTemplate.update(sql,
                user.getUserId(),
                user.getFirstName(),
                user.getLastName(),
                user.getFullName(),
                user.getEmail(),
                user.getMobile(),
                user.getAddress(),
                user.getPassword(),
                user.getRole());
    }

    /**
     * Validate login credentials.
     * @return User if credentials match, null otherwise
     */
    public User validateLogin(String userId, String password) {
        try {
            String sql = "SELECT * FROM tbl_UserProfile WHERE user_id = ? AND password = ?";
            return jdbcTemplate.queryForObject(sql, userRowMapper, userId, password);
        } catch (EmptyResultDataAccessException e) {
            return null;
        }
    }
}

            String userId = scanner.nextLine().trim();

            if (userId.equalsIgnoreCase("exit")) {
                System.out.println("\n  Goodbye! Thank you for using Nimbus.\n");
                break;
            }

            System.out.print("  Password: ");
            String password = scanner.nextLine().trim();

            User user = userDAO.validateLogin(userId, password);

            if (user == null) {
                System.out.println("\n  ✗ Invalid credentials. Try again or type 'exit' to quit.\n");
                continue;
            }

            System.out.println("\n  ✓ Login successful! Role: " + user.getRole());

            // Route to role-based menu
            switch (user.getRole().toLowerCase()) {
                case "customer":
                    customerLoop(user);
                    break;
                case "officer":
                    officerLoop(user);
                    break;
                case "support":
                    supportLoop(user);
                    break;
                default:
                    System.out.println("  Unknown role: " + user.getRole());

        }

    private void bookParcel(User user) {
        System.out.println("\n  --- Book a Parcel ---\n");

        System.out.print("  Receiver Name: ");
        String receiverName = scanner.nextLine().trim();
        if (!InputValidator.isValidName(receiverName)) {
            System.out.println("  ✗ Invalid name. Only letters and spaces allowed.");
            return;
        }

        System.out.print("  Receiver Address: ");
        String receiverAddress = scanner.nextLine().trim();

        System.out.print("  Receiver Mobile: ");
        String receiverMobile = scanner.nextLine().trim();
        if (!InputValidator.isValidMobile(receiverMobile)) {
            System.out.println("  ✗ Invalid mobile number. Must be 10 digits starting with 6-9.");
            return;
        }

        System.out.print("  Weight (grams): ");
        double weight;
        try {
            weight = Double.parseDouble(scanner.nextLine().trim());
        } catch (NumberFormatException e) {
            System.out.println("  ✗ Invalid weight. Enter a number.");
            return;
        }
        if (!InputValidator.isValidWeight(weight)) {
            System.out.println("  ✗ Weight must be between 1g and 50,000g.");
            return;
        }

        System.out.print("  Contents Description: ");
        String contents = scanner.nextLine().trim();

        System.out.println("\n" + PricingCalculator.getDeliveryOptions());
        System.out.print("\n  Choose delivery type (1/2/3): ");
        String typeChoice = scanner.nextLine().trim();
        String deliveryType;
        switch (typeChoice) {
            case "1": deliveryType = "Economy"; break;
            case "2": deliveryType = "Standard"; break;
            case "3": deliveryType = "Express"; break;
            default:
                System.out.println("  ✗ Invalid delivery type.");
                return;
        }

        double cost = PricingCalculator.calculateCost(weight, deliveryType);

        Booking booking = new Booking();

        booking.setContentsDescription(contents);

        booking.setServiceCost(cost);
